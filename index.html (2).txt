  <script>
    var userInput = document.getElementById("userInput");
    var loader = document.getElementById("loader");
    var outputContainer = document.getElementById("outputContainer");
    var output = document.getElementById("output");
    var micBtn = document.getElementById("micBtn");
    var micIcon = document.getElementById("micIcon");
    var micStatus = document.getElementById("micStatus");
    var autoSpeakCheckbox = document.getElementById("autoSpeak");
    var voiceSelect = document.getElementById("voiceSelect");
    
    var recognition = null;
    var isRecording = false;
    var audioCtx = null;
    var currentSource = null;

    // ========================================================
    // ЗМІНИ ТУТ: Встав своє посилання Web App URL всередину лапок
    // ========================================================
    var scriptUrl = "https://script.google.com/macros/library/d/1wkyys4_th5yf1h70NpHamOOTX4llEsqqMtrHD7KJS4H7xzP8qMO4zWSt/53";
    
    function initAudioContext() {
      if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); }
      if (audioCtx.state === "suspended") { audioCtx.resume(); }
    }
    
    function setupSpeechRecognition() {
      if (recognition) return true;
      var SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
      if (!SpeechRecognition) {
        micStatus.innerText = "Голос недоступний";
        micBtn.disabled = true;
        return false;
      }
      
      recognition = new SpeechRecognition();
      recognition.continuous = false;
      recognition.interimResults = false;
      recognition.lang = "uk-UA";
      
      recognition.onstart = function() {
        isRecording = true;
        micBtn.classList.add("recording-active");
        micIcon.innerText = "mic_off";
        micStatus.innerText = "Слухаю...";
      };
      
      recognition.onresult = function(e) {
        var t = e.results[0][0].transcript;
        userInput.value = userInput.value.trim() ? userInput.value + " " + t : t;
      };
      
      recognition.onerror = function() { stopRecording(); };
      recognition.onend = function() { stopRecording(); };
      return true;
    }
    
    function toggleSpeech() {
      initAudioContext(); stopAudio();
      if (!setupSpeechRecognition()) return;
      if (isRecording) { recognition.stop(); } else { try { recognition.start(); } catch(e){} }
    }
    
    function stopRecording() {
      isRecording = false;
      micBtn.classList.remove("recording-active");
      micIcon.innerText = "mic";
      micStatus.innerText = "Натисніть для запису";
    }
    
    function stopAudio() { if (currentSource) { try { currentSource.stop(); } catch(e){} currentSource = null; } }
    function clearInput() { userInput.value = ""; stopAudio(); }
    
    function handleAction(type) {
      initAudioContext(); var text = userInput.value.trim(); if (!text) return; showLoader(true);
      fetch(scriptUrl, { method: "POST", body: JSON.stringify({ action: type, text: text }) })
        .then(r => r.json())
        .then(j => { showLoader(false); displayResult(j.result); if(type==="save") userInput.value=""; })
        .catch((e) => { console.error(e); showLoader(false); displayResult("❌ Помилка зв'язку з Google API"); });
    }

    function speakText(text) {
      stopAudio(); initAudioContext();
      fetch(scriptUrl, { method: "POST", body: JSON.stringify({ action: "tts", text: text, voice: voiceSelect.value }) })
        .then(r => r.json()).then(j => { processAndPlay(j.audio, text); });
    }
    
    function processAndPlay(b64, rawText) {
      if (b64.indexOf("FALLBACK:") === 0) {
        var u = new SpeechSynthesisUtterance(decodeURIComponent(b64.replace("FALLBACK:", ""))); u.lang = "ru-RU"; window.speechSynthesis.speak(u);
        return;
      }
      var binaryString = window.atob(b64);
      var bytes = new Uint8Array(binaryString.length);
      for (var i = 0; i < binaryString.length; i++) { bytes[i] = binaryString.charCodeAt(i); }
      audioCtx.decodeAudioData(bytes.buffer, function(buffer) {
        stopAudio();
        currentSource = audioCtx.createBufferSource();
        currentSource.buffer = buffer;
        currentSource.connect(audioCtx.destination);
        currentSource.start(0);
      });
    }
    
    function speakOutput() { initAudioContext(); speakText(output.innerText); }
    function showLoader(show) { if (show) { loader.classList.remove("hidden"); outputContainer.classList.add("hidden"); stopAudio(); } else { loader.classList.add("hidden"); } }
    function displayResult(text) { output.innerText = text; outputContainer.classList.remove("hidden"); if (autoSpeakCheckbox.checked) speakText(text); }
    
    window.addEventListener("DOMContentLoaded", setupSpeechRecognition);
  </script>
